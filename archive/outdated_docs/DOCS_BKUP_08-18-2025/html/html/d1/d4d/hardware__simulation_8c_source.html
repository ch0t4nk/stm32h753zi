342 static simulation_error_t
343 execute_python_command(const char *command, char *output, size_t output_size) {
344  char full_command[1024];
345  FILE *pipe;
346 
347  /* Build full Python command */
348  snprintf(full_command, sizeof(full_command),
349  "python3 -c \"%s\" 2>/dev/null", command);
350 
351  /* Execute command and capture output */
352  pipe = popen(full_command, "r");
353  if (!pipe) {
354  return SIM_ERROR_COMMAND_FAILED;
355  }
356 
357  if (fgets(output, output_size, pipe) == NULL) {
358  pclose(pipe);
359  return SIM_ERROR_COMMAND_FAILED;
360  }
361 
362  pclose(pipe);
363 
364  /* Remove trailing newline */
365  size_t len = strlen(output);
366  if (len > 0 && output[len - 1] == '\n') {
367  output[len - 1] = '\0';
368  }
369 
370  return SIM_OK;
371 }
