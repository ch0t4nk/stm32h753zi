215  if (efficiency_change > 0.1f) {

218  } else if (efficiency_change < -0.1f) {
219  // Efficiency degraded - reverse direction
220  alg_data->current_setpoint_pct -= alg_data->learning_rate;
221  alg_data->learning_rate *= 0.8f; // Reduce learning rate for stability
222  } else {
223  // Minimal change - reduce learning rate and fine-tune
224  alg_data->learning_rate *= 0.95f;
225  }
226 
227  // Apply safety limits
228  if (alg_data->current_setpoint_pct > 100.0f + MAX_CURRENT_INCREASE_PCT) {
229  alg_data->current_setpoint_pct = 100.0f + MAX_CURRENT_INCREASE_PCT;
230  }
231  if (alg_data->current_setpoint_pct < 50.0f) {
232  alg_data->current_setpoint_pct = 50.0f; // Minimum 50% current
233  }
234 
235  // Check convergence

237  alg_data->converged = true;
238  context->results.optimization_converged = true;

719 SystemError_t run_optimization_simulation(motor_id_t motor_id,
720  uint8_t scenario_id) {

723  }

730

733 
734  // Configure optimization for simulation scenario
735  OptimizationConfig_t config = {
736  .algorithm = EFFICIENCY_ALG_ADAPTIVE_CURRENT,
737  .target = OPT_TARGET_POWER_CONSUMPTION,
738  .target_efficiency_pct =
739  scenario->optimization_potential * 0.8f, // 80% of potential
740  .convergence_threshold = DEFAULT_CONVERGENCE_THRESHOLD,
741  .max_iterations = scenario->convergence_iterations,
742  .characterization_time_ms = 1000, // Fast simulation
743  .safety_limits_enabled = true,
744  .real_time_adaptation = true,
745  .adaptation_rate = 0.2f, // Faster learning for simulation
746  .validation_time_ms = 500};
747 
748  // Configure and start optimization
749  SystemError_t result = configure_motor_optimization(motor_id, &config);
750  if (result != SYSTEM_OK) {
751  return result;
752  }
753 
754  result = start_motor_optimization(motor_id);
755  if (result != SYSTEM_OK) {
756  return result;
757  }
758 
759  // Simulate optimization process
760  for (uint32_t i = 0; i < scenario->convergence_iterations; i++) {
761  result = update_motor_optimization(motor_id, 0.02f); // 20ms updates
762  if (result != SYSTEM_OK) {
763  break;
764  }
765 
766  // Check if optimization converged early
767  if (optimization_contexts[motor_id].results.optimization_converged) {
768  break;
769  }
770  }
771 
772  return result;
773 }
