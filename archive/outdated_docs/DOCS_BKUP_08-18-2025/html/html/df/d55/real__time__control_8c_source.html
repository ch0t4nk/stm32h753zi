405 static void execute_priority_tasks(RTTaskPriority_t priority) {
406  uint32_t current_time = get_microsecond_timestamp();

425 static void execute_task(RTTask_t *task, uint32_t current_time) {
426  // Check for deadline miss
427  uint32_t elapsed = current_time - task->last_execution;
428  if (elapsed > task->deadline_us && task->execution_count > 0) {
429  task->missed_deadlines++;
430  }
431 
432  // Record execution start
433  uint32_t execution_start = get_microsecond_timestamp();
434  task->state = RT_TASK_RUNNING;
435 
436  // Execute task function
437  if (task->function != NULL) {
438  task->function(task->context);
439  }
440 
441  // Record execution time
442  uint32_t execution_time = get_microsecond_timestamp() - execution_start;
443 
444  task->last_execution = current_time;
445  task->execution_count++;
446  task->total_execution_time += execution_time;
447  task->state = RT_TASK_READY;
448 
449  // Update maximum execution time
450  if (execution_time > task->max_execution_time) {
451  task->max_execution_time = execution_time;
452  }
453 }

458 static void update_timing_statistics(uint32_t execution_time) {
459  // Check for timing overrun
460  if (execution_time > RT_CONTROL_LOOP_PERIOD_US) {
461  rt_control_system.timing.overrun_count++;
462  }
463 
464  // Calculate jitter
465  static uint32_t last_execution_time = 0;
466  if (last_execution_time > 0) {
467  uint32_t jitter = (execution_time > last_execution_time)
468  ? (execution_time - last_execution_time)
469  : (last_execution_time - execution_time);
470 
471  if (jitter > rt_control_system.timing.max_jitter_us) {
472  rt_control_system.timing.max_jitter_us = jitter;
473  }
474  }
475  last_execution_time = execution_time;
476 }
