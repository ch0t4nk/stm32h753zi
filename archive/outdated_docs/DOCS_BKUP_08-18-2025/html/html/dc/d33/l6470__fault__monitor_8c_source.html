143 void l6470_fault_monitor_flag_interrupt(void) {
144  uint32_t interrupt_start_time = HAL_GetTick();
145 
146  // FLAG pin is active low - fault when low
147  GPIO_PinState flag_state =
148  HAL_GPIO_ReadPin(MOTOR_FLAG_PORT, MOTOR_FLAG_PIN);
149 
150  if (flag_state == GPIO_PIN_RESET) {
151  // Fault detected - read L6470 status to determine fault type
152  uint16_t status_register = 0;
153 
154  // Read status from both motors (assuming dual motor setup)
155  for (uint8_t motor_id = 0; motor_id < MAX_MOTORS; motor_id++) {
156  SystemError_t result =
157  l6470_get_status(motor_id, &status_register);
158  if (result == SYSTEM_OK) {
159  L6470HwFaultType_t fault_type =
160  decode_l6470_status(status_register);
161  if (fault_type != L6470_HW_FAULT_NONE) {
162  process_l6470_fault(fault_type);
163  }
164  }
165  }
166  }
167 
168  // Record reaction time for safety validation
169  fault_status.fault_reaction_time_us =
170  (HAL_GetTick() - interrupt_start_time) * 1000;
171  fault_status.fault_count++;
172 }

263 static void process_l6470_fault(L6470HwFaultType_t fault_type) {
264  fault_status.fault_active = true;
265  fault_status.active_faults |= fault_type;
266  fault_status.last_fault_time = HAL_GetTick();
267 
268  // Determine response based on fault severity
269  switch (fault_type) {
270  case L6470_HW_FAULT_OVERCURRENT:
271  case L6470_HW_FAULT_THERMAL_SHUTDOWN:
272  case L6470_HW_FAULT_STALL_DETECTED:
273  // Critical faults - immediate emergency stop
274  execute_emergency_stop(ESTOP_SOURCE_OVERCURRENT);
275  break;
276 
277  case L6470_HW_FAULT_THERMAL_WARNING:
278  case L6470_HW_FAULT_UNDERVOLTAGE:
279  // Warning conditions - log but continue if safe
280  fault_monitor_record_system_fault(SYSTEM_FAULT_POWER_FAILURE,
281  FAULT_SEVERITY_WARNING, fault_type);
282  break;
283 
284  case L6470_HW_FAULT_STEP_LOSS:
285  case L6470_HW_FAULT_COMMAND_ERROR:
286  // Operational faults - stop affected motor
287  // Implementation depends on motor identification
288  break;
289 
290  default:
291  // Unknown fault - treat as critical
292  execute_emergency_stop(ESTOP_SOURCE_FAULT_CASCADE);
293  break;
294  }
295 }

300 static L6470HwFaultType_t decode_l6470_status(uint16_t status_register) {
301  L6470HwFaultType_t fault_type = L6470_HW_FAULT_NONE;
302 
303  // Check status bits (based on L6470 datasheet)
304  if (status_register & L6470_STATUS_OCD) {
305  fault_type |= L6470_HW_FAULT_OVERCURRENT;
306  }
307  if (status_register & L6470_STATUS_TH_SD) {
308  fault_type |= L6470_HW_FAULT_THERMAL_SHUTDOWN;
309  }
310  if (status_register & L6470_STATUS_TH_WRN) {
311  fault_type |= L6470_HW_FAULT_THERMAL_WARNING;
312  }
313  if (status_register & L6470_STATUS_UVLO) {
314  fault_type |= L6470_HW_FAULT_UNDERVOLTAGE;
315  }
316  if (status_register &
317  (L6470_STATUS_STEP_LOSS_A | L6470_STATUS_STEP_LOSS_B)) {
318  fault_type |= L6470_HW_FAULT_STEP_LOSS;
319  }
320  if (status_register &
321  (L6470_STATUS_WRONG_CMD | L6470_STATUS_NOTPERF_CMD)) {
322  fault_type |= L6470_HW_FAULT_COMMAND_ERROR;
323  }
324 
325  return fault_type;
326 }
