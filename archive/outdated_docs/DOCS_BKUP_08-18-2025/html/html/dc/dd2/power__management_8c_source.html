194 SystemError_t power_management_update(void) {
195  if (!power_mgmt_initialized) {
196  return ERROR_NOT_INITIALIZED;
197  }
198 
199  if (xSemaphoreTake(power_mgmt_mutex, pdMS_TO_TICKS(10)) != pdTRUE) {
200  return ERROR_TIMEOUT;
201  }
202 
203  uint32_t current_time = xTaskGetTickCount();
204 
205  // Update idle duration
206  power_state.idle_duration_ms =
207  current_time - power_state.last_activity_time;
208 
209  // Evaluate potential power mode transition
210  PowerMode_t recommended_mode = evaluate_power_transition();
211 
212  if (recommended_mode != power_state.current_mode &&
213  !power_state.mode_transition_pending) {
214 
215  // Validate transition safety
216  if (validate_power_transition_safety(power_state.current_mode,
217  recommended_mode)) {
218  power_state.requested_mode = recommended_mode;
219  power_state.mode_transition_pending = true;
220 
221  printf("Power Management: Transition recommended: %d -> %d (idle: "
222  "%lu ms)\r\n",
223  power_state.current_mode, recommended_mode,
224  power_state.idle_duration_ms);
225  }
226  }
227 
228  // Execute pending transition if requested
229  if (power_state.mode_transition_pending &&
230  power_state.requested_mode != power_state.current_mode) {
231 
232  SystemError_t result = apply_power_mode(power_state.requested_mode);
233  if (result == SYSTEM_OK) {
234  PowerMode_t old_mode = power_state.current_mode;
235  power_state.current_mode = power_state.requested_mode;
236  power_state.mode_entry_time = current_time;
237  power_state.mode_transitions_count++;
238  power_state.mode_transition_pending = false;
239 
240  update_power_statistics(old_mode, power_state.current_mode);
241 
242  printf("Power Management: Mode transition complete: %d -> %d\r\n",
243  old_mode, power_state.current_mode);
244  } else {
245  printf("Power Management: Mode transition failed: %d\r\n", result);
246  power_state.mode_transition_pending = false;
247  }
248  }
249 
250  xSemaphoreGive(power_mgmt_mutex);
251  return SYSTEM_OK;

444 static PowerMode_t evaluate_power_transition(void) {
445  uint32_t idle_time = power_state.idle_duration_ms;
446 
447  // Evaluate based on idle time and current mode
448  switch (power_state.current_mode) {
449  case POWER_MODE_ACTIVE:
450  if (idle_time > POWER_IDLE_TIMEOUT_MS) {
451  return POWER_MODE_QUIET;
452  }
453  break;
454 
455  case POWER_MODE_QUIET:
456  if (idle_time < 5000) { // 5 seconds hysteresis
457  return POWER_MODE_ACTIVE;
458  } else if (idle_time > (POWER_IDLE_TIMEOUT_MS * 2)) {
459  return POWER_MODE_IDLE;
460  }
461  break;
462 
463  case POWER_MODE_IDLE:
464  if (idle_time < 10000) { // 10 seconds hysteresis
465  return POWER_MODE_QUIET;
466  } else if (idle_time > POWER_SLEEP_TIMEOUT_MS) {
467  return POWER_MODE_SLEEP;
468  }
469  break;
470 
471  case POWER_MODE_SLEEP:
472  if (idle_time < 30000) { // 30 seconds hysteresis
473  return POWER_MODE_IDLE;
474  }
475  break;
476 
477  default:
478  return POWER_MODE_ACTIVE;
479  }
480 
481  return power_state.current_mode; // No change
