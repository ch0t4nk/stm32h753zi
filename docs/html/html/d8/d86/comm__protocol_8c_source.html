606 static uint16_t calculate_message_checksum(const MessageHeader_t* header, const uint8_t* payload) {
607  // Simple CRC16 implementation
608  uint16_t crc = 0xFFFF;
609 
610  // Include header in checksum (excluding checksum field itself)
611  const uint8_t* data = (const uint8_t*)header;
612  for (size_t i = 0; i < offsetof(MessageHeader_t, checksum); i++) {
613  crc ^= data[i];
614  for (int j = 0; j < 8; j++) {
615  if (crc & 1) {
616  crc = (crc >> 1) ^ 0xA001;
617  } else {
618  crc >>= 1;
619  }
620  }
621  }
622 
623  // Include payload in checksum
624  for (uint16_t i = 0; i < header->payload_length; i++) {
625  crc ^= payload[i];
626  for (int j = 0; j < 8; j++) {
627  if (crc & 1) {
628  crc = (crc >> 1) ^ 0xA001;
629  } else {
630  crc >>= 1;
631  }
632  }
633  }
634 
635  return crc;
636 }
