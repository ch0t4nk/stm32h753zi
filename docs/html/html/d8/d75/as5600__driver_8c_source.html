378 SystemError_t as5600_calibrate_zero(uint8_t encoder_id, float current_angle) {
379  SystemError_t result = as5600_validate_encoder_id(encoder_id);
380  if (result != SYSTEM_OK) {
381  return result;
382  }
383 
384  // Read current raw angle
385  uint16_t raw_angle;
386  result = as5600_read_raw_angle(encoder_id, &raw_angle);
387  if (result != SYSTEM_OK) {
388  return result;
389  }
390 
391  // Calculate zero offset
392  float raw_degrees = as5600_raw_to_degrees(raw_angle);
393  float offset_degrees = raw_degrees - current_angle;
394 
395  // Normalize offset to 0-360 range
396  while (offset_degrees < 0.0f)
397  offset_degrees += 360.0f;
398  while (offset_degrees >= 360.0f)
399  offset_degrees -= 360.0f;
400

403  (uint16_t)((offset_degrees / 360.0f) * 4095.0f + 0.5f);
404 
405  // Write zero position to AS5600 (requires PROG pin low - not implemented)
406  // For now, just store in state for software compensation
407  encoder_state[encoder_id].calibrated = true;
408  encoder_state[encoder_id].zero_offset =
409  zero_position; // Use the calculated value

549 static SystemError_t as5600_i2c_read_16bit(uint8_t encoder_id,
550  uint8_t reg_addr_high,
551  uint16_t *value) {
552  AS5600_EncoderState_t *state = &encoder_state[encoder_id];
553 
554 #if SIMULATION_ENABLED
555  if (state->simulation_mode) {
556  // Use simulation backend for 16-bit reads
557  uint8_t high_byte, low_byte;
558  bool sim_result;
559 
560  // Read high byte
561  sim_result =
562  as5600_sim_read_register(encoder_id, reg_addr_high, &high_byte);
563  if (!sim_result) {
564  return ERROR_ENCODER_COMMUNICATION;
565  }
566 
567  // Read low byte
568  sim_result =
569  as5600_sim_read_register(encoder_id, reg_addr_high + 1, &low_byte);
570  if (!sim_result) {
571  return ERROR_ENCODER_COMMUNICATION;
572  }
573 
574  // Combine high and low bytes (MSB first)
575  *value = ((uint16_t)high_byte << 8) | low_byte;
576  return SYSTEM_OK;
577  }
578 #endif
579 
580  uint8_t data[2];
581 
582  HAL_I2C_Transaction_t transaction = {.device_address = state->i2c_address,
583  .register_address = reg_addr_high,
584  .data = data,
585  .data_size = 2,
586  .timeout_ms = AS5600_I2C_TIMEOUT,
587  .use_register_address = true};
588 
589  // Read 2 bytes starting from high byte register
590  SystemError_t result = HAL_Abstraction_I2C_MemRead(
591  encoder_id == 0 ? HAL_I2C_INSTANCE_1 : HAL_I2C_INSTANCE_2,
592  &transaction);
593 
594  if (result != SYSTEM_OK) {
595  return ERROR_ENCODER_COMMUNICATION;
596  }
597 
598  // Combine high and low bytes (MSB first)
599  *value = ((uint16_t)data[0] << 8) | data[1];
600 
601  return SYSTEM_OK;
602 }

672 static void as5600_calculate_velocity(uint8_t encoder_id) {
673  AS5600_EncoderState_t *state = &encoder_state[encoder_id];
674  uint32_t current_time = HAL_Abstraction_GetTick();
675 
676  if (state->read_count <= 1) {
677  state->velocity_dps = 0.0f;
678  return;
679  }
680 
681  // Calculate time difference
682  uint32_t time_diff = current_time - state->last_read_time;
683  if (time_diff == 0) {
684  return; // Avoid division by zero
685  }
686 
687  // Calculate angle difference (handle 360/0 degree wraparound)
688  float angle_diff = state->angle_degrees - state->previous_angle;
689 
690  // Handle wraparound
691  if (angle_diff > 180.0f) {
692  angle_diff -= 360.0f;
693  } else if (angle_diff < -180.0f) {
694  angle_diff += 360.0f;
695  }
696 
697  // Calculate velocity in degrees per second
698  float time_diff_s = (float)time_diff / 1000.0f;
699  state->velocity_dps = angle_diff / time_diff_s;
700 }
