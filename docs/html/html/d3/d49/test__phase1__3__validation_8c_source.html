205 
206  test_packet.timestamp_us = (uint32_t)(time * 1000000);
207  test_packet.encoder_position_deg = response;
208  test_packet.data_valid = true;
209 
210  test_dataset.samples[test_dataset.sample_count] = test_packet;
211  test_dataset.sample_count++;
212  }
213 
214  // Simple time constant identification (find 63% of final value)

216  float target_value = final_value * 0.632f; // 63.2% for time constant
217 
218  float identified_time = 0.0f;

220  if (test_dataset.samples[i].encoder_position_deg >= target_value) {
