125  FAULT_SEVERITY_CRITICAL, l6470_faults);

128  fault_monitor_record_motor_fault(motor_id, MOTOR_FAULT_OVERCURRENT,
129  FAULT_SEVERITY_CRITICAL, l6470_faults);
130  }
131  if (l6470_faults & L6470_FAULT_THERMAL_SHUTDOWN) {
132  fault_monitor_record_motor_fault(motor_id, MOTOR_FAULT_OVERTEMPERATURE,
133  FAULT_SEVERITY_CRITICAL, l6470_faults);
134  }
135  if (l6470_faults & L6470_FAULT_THERMAL_WARNING) {
136  fault_monitor_record_motor_fault(motor_id, MOTOR_FAULT_TEMP_WARNING,
137  FAULT_SEVERITY_WARNING, l6470_faults);
138  }
139  if (l6470_faults & L6470_FAULT_UVLO) {
140  fault_monitor_record_motor_fault(motor_id, MOTOR_FAULT_UNDERVOLTAGE,
141  FAULT_SEVERITY_CRITICAL, l6470_faults);
142  }
143  if (l6470_faults & L6470_FAULT_STALL_DETECTION) {
144  fault_monitor_record_motor_fault(motor_id, MOTOR_FAULT_POSITION_ERROR,
145  FAULT_SEVERITY_ERROR, l6470_faults);

147  }
148  }

621  uint32_t motor_id, uint32_t additional_data) {
622  // Look for existing fault record
623  FaultRecord_t* existing = find_fault_record(fault_code);
624 
625  if (existing != NULL) {
626  // Update existing record
627  existing->count++;
628  existing->timestamp = HAL_GetTick();
629  existing->additional_data = additional_data;
630  existing->active = true;
631  existing->acknowledged = false;
632  return SYSTEM_OK;
633  }
634 
635  // Add new fault record
636  if (total_fault_records < MAX_FAULT_RECORDS) {
637  fault_record_index = total_fault_records;
638  total_fault_records++;
639  } else {
640  // Circular buffer - overwrite oldest
641  fault_record_index = (fault_record_index + 1) % MAX_FAULT_RECORDS;
642  }
643 
644  FaultRecord_t* record = &fault_records[fault_record_index];
645  record->fault_code = fault_code;
646  record->severity = severity;
647  record->timestamp = HAL_GetTick();
648  record->count = 1;
649  record->motor_id = motor_id;
650  record->additional_data = additional_data;
651  record->active = true;
652  record->acknowledged = false;
653 
654  return SYSTEM_OK;
655 }
