114  const TickType_t xFrequency = pdMS_TO_TICKS(MOTOR_CONTROL_PERIOD_MS);
115  MotorCommand_t received_command;
116 
117  // Initialize the xLastWakeTime variable with the current time
118  xLastWakeTime = xTaskGetTickCount();
119 
120  printf("MotorControlTask: Started with %dms period (Priority %d)\r\n",
121  MOTOR_CONTROL_PERIOD_MS, MOTOR_CONTROL_TASK_PRIORITY);
122 
123  for (;;) {
124  motor_task_cycles++;
125 
126  // Check for emergency stop
127  if (emergency_stop_active) {
128  // During emergency stop, just wait and don't execute motor control
129  vTaskDelayUntil(&xLastWakeTime, xFrequency);
130  continue;
131  }
132 
133  // Process any pending motor commands (non-blocking)
134  while (xQueueReceive(motor_command_queue, &received_command, 0) ==
135  pdTRUE) {
136  // Process the motor command using existing controller
137  printf("MotorControlTask: Processing command for motor %d\r\n",
138  received_command.motor_id);
139  // TODO: Integrate with motor_controller_set_target() or similar
140  // function
141  }
142 
143  // Execute main motor control loop using existing function
144  SystemError_t result;
145 
146  // Protect motor state with mutex
147  if (xSemaphoreTake(motor_state_mutex, pdMS_TO_TICKS(5)) == pdTRUE) {
148  result = motor_controller_update();
149  xSemaphoreGive(motor_state_mutex);
150  } else {
151  result = ERROR_BUSY;
152  printf(
153  "MotorControlTask: Failed to acquire motor state mutex\r\n");
154  }
155 
156  // Handle any errors from motor control
157  if (result != SYSTEM_OK) {
158  SafetyEvent_t safety_event = {
159  .event_type = SAFETY_EVENT_MOTOR_ERROR,
160  .error_code = result,
161  .timestamp = xTaskGetTickCount(),
162  .source_task = xTaskGetCurrentTaskHandle()};
163 
164  // Send to safety monitor (non-blocking to prevent task blocking)
165  if (xQueueSend(safety_event_queue, &safety_event, 0) != pdTRUE) {
166  printf("MotorControlTask: Safety event queue full!\r\n");
167  }
168  }
169 
170  // Performance monitoring (every 1000 cycles = 1 second)
171  if ((motor_task_cycles % 1000) == 0) {
172  UBaseType_t uxHighWaterMark = uxTaskGetStackHighWaterMark(NULL);
173  printf("MotorControlTask: Cycles=%lu, Stack Free=%lu words\r\n",
174  motor_task_cycles, uxHighWaterMark);
175  }
176

178  vTaskDelayUntil(&xLastWakeTime, xFrequency);

191  SafetyEvent_t received_event;
192 
193  // Initialize the xLastWakeTime variable with the current time
194  xLastWakeTime = xTaskGetTickCount();
195 
196  printf("SafetyMonitorTask: Started with %dms period (Priority %d - "
197  "HIGHEST)\r\n",
198  SAFETY_CHECK_PERIOD_MS, SAFETY_MONITOR_TASK_PRIORITY);
199 
200  for (;;) {
201  safety_task_cycles++;
202 
203  // Process any safety events from other tasks (non-blocking)
204  while (xQueueReceive(safety_event_queue, &received_event, 0) ==
205  pdTRUE) {
206  printf("SafetyMonitorTask: Received safety event type %d from "
207  "task 0x%p\r\n",
208  received_event.event_type, received_event.source_task);
209 
210  // Handle critical safety events immediately
211  if (received_event.event_type == SAFETY_EVENT_EMERGENCY_STOP) {
212  emergency_stop_active = true;
213  printf("SafetyMonitorTask: EMERGENCY STOP ACTIVATED!\r\n");
214 
215  // Suspend motor control task during emergency
216  if (motor_control_task_handle != NULL) {
217  vTaskSuspend(motor_control_task_handle);
218  printf(
219  "SafetyMonitorTask: Motor control task suspended\r\n");
220  }
221  }
222  }
223 
224  // Execute periodic safety checks using existing function
225  SystemError_t safety_result = safety_system_task();
226 
227  // Handle safety system results
228  if (safety_result == ERROR_SAFETY_EMERGENCY_STOP) {
229  if (!emergency_stop_active) {
230  emergency_stop_active = true;
231  printf("SafetyMonitorTask: Safety system triggered emergency "
232  "stop\r\n");
233 
234  // Suspend motor control during emergency
235  if (motor_control_task_handle != NULL) {
236  vTaskSuspend(motor_control_task_handle);
237  }
238  }
239  } else if (safety_result == SYSTEM_OK && emergency_stop_active) {
240  // Safety system cleared - resume normal operation
241  emergency_stop_active = false;
242  printf("SafetyMonitorTask: Emergency stop cleared - resuming "
243  "operation\r\n");
244 
245  // Resume motor control task
246  if (motor_control_task_handle != NULL) {
247  vTaskResume(motor_control_task_handle);
248  }
249  } else if (safety_result != SYSTEM_OK) {
250  printf("SafetyMonitorTask: Safety check warning (code: %d)\r\n",
251  safety_result);
252  }
253 
254  // Performance monitoring (every 500 cycles = 1 second at 500Hz)
255  if ((safety_task_cycles % 500) == 0) {
256  UBaseType_t uxHighWaterMark = uxTaskGetStackHighWaterMark(NULL);
257  printf("SafetyMonitorTask: Cycles=%lu, Stack Free=%lu words, "
258  "Emergency=%s\r\n",
259  safety_task_cycles, uxHighWaterMark,
260  emergency_stop_active ? "ACTIVE" : "CLEARED");
261  }
262 
263  // Wait for the next cycle (precise 2ms timing)
264  vTaskDelayUntil(&xLastWakeTime, xFrequency);
265  }
