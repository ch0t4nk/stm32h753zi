300 SystemError_t motor_controller_update(void) {
301  if (!motor_controller_initialized) {
302  return ERROR_MOTOR_INIT_FAILED;
303  }
304 
305  SystemError_t overall_result = SYSTEM_OK;
306 
307  // Update all motors
308  for (uint8_t motor_id = 0; motor_id < MAX_MOTORS; motor_id++) {
309  MotorControlState_t* state = &motor_control_state[motor_id];
310 
311  if (!state->enabled) {
312  continue;
313  }
314 
315  // Update position from encoder
316  SystemError_t result = motor_controller_update_position(motor_id);
317  if (result != SYSTEM_OK) {
318  state->fault_count++;
319  overall_result = result;
320  continue;
321  }
322 
323  // Calculate position error
324  state->position_error_deg = motor_controller_calculate_position_error(
325  state->target_position_deg, state->current_position_deg);
326 
327  // Check if movement is complete (within tolerance)
328  if (state->state == MOTOR_STATE_RUNNING &&
329  fabsf(state->position_error_deg) < POSITION_TOLERANCE_DEG) {
330  state->state = MOTOR_STATE_IDLE;
331  }
332 
333  // Perform safety checks
334  result = motor_controller_safety_check(motor_id);
335  if (result != SYSTEM_OK) {
336  state->fault_count++;
337  overall_result = result;
338  // Emergency stop on safety fault
339  motor_controller_emergency_stop(motor_id);
340  }
341 
342  state->last_update_time = HAL_GetTick();
343  }
344 
345  return overall_result;
346 }

549 static float motor_controller_calculate_position_error(float target, float current) {
550  float error = target - current;
551 
552  // Handle wraparound (shortest path)
553  while (error > 180.0f) {
554  error -= 360.0f;
555  }
556  while (error < -180.0f) {
557  error += 360.0f;
558  }
559 
560  return error;
561 }
