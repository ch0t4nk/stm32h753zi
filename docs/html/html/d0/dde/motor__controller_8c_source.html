312 SystemError_t motor_controller_update(void) {
313  if (!motor_controller_initialized) {
314  return ERROR_MOTOR_INIT_FAILED;
315  }
316 
317  SystemError_t overall_result = SYSTEM_OK;
318 
319  // Update all motors
320  for (uint8_t motor_id = 0; motor_id < MAX_MOTORS; motor_id++) {
321  MotorControlState_t *state = &motor_control_state[motor_id];
322 
323  if (!state->enabled) {
324  continue;
325  }
326 
327  // Update position from encoder
328  SystemError_t result = motor_controller_update_position(motor_id);
329  if (result != SYSTEM_OK) {
330  state->fault_count++;
331  overall_result = result;
332  continue;
333  }
334 
335  // Calculate position error
336  state->position_error_deg = motor_controller_calculate_position_error(
337  state->target_position_deg, state->current_position_deg);
338 
339  // Check if movement is complete (within tolerance)
340  if (state->state == MOTOR_STATE_RUNNING &&
341  fabsf(state->position_error_deg) < POSITION_TOLERANCE_DEG) {
342  state->state = MOTOR_STATE_IDLE;
343  }
344 
345  // Perform safety checks
346  result = motor_controller_safety_check(motor_id);
347  if (result != SYSTEM_OK) {
348  state->fault_count++;
349  overall_result = result;
350  // Emergency stop on safety fault
351  motor_controller_emergency_stop(motor_id);
352  }
353 
354  state->last_update_time = HAL_Abstraction_GetTick();
355  }
356 
357  return overall_result;
358 }

570 static float motor_controller_calculate_position_error(float target,
571  float current) {
572  float error = target - current;
573 
574  // Handle wraparound (shortest path)
575  while (error > 180.0f) {
576  error -= 360.0f;
577  }
578  while (error < -180.0f) {
579  error += 360.0f;
580  }
581 
582  return error;
583 }
