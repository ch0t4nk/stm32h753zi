137 SystemError_t emergency_stop_process(void) {
138  if (!estop_context.initialized) {
139  return ERROR_NOT_INITIALIZED;
140  }
141 
142  SystemError_t result;
143 
144  // Process button debouncing
145  result = estop_process_button_debounce();
146  if (result != SYSTEM_OK) {
147  return result;
148  }
149 
150  // State machine processing
151  switch (estop_context.state) {
152  case EMERGENCY_STOP_ARMED:
153  // Check for button press
154  if (estop_context.button_debounced) {
155  result = emergency_stop_execute(ESTOP_SOURCE_BUTTON);
156  }
157  break;
158 
159  case EMERGENCY_STOP_TRIGGERED:
160  // Already triggered, maintain safety state
161  break;
162 
163  case EMERGENCY_STOP_RESET_PENDING: {
164  // Check if sufficient time has passed for reset
165  uint32_t current_time = HAL_Abstraction_GetTick();
166  if ((current_time - estop_context.state_entry_time) >
167  1000) { // 1 second delay
168  // Deactivate safety relays
169  result = estop_activate_safety_relays(false);
170  if (result == SYSTEM_OK) {
171  estop_change_state(EMERGENCY_STOP_ARMED);
172  }
173  }
174  break;
175  }
176 
177  default:
178  estop_change_state(EMERGENCY_STOP_FAULT);
179  return ERROR_INVALID_STATE;
180  }
181 
182  // Update LED indication
183  result = estop_update_led_indication();
184 
185  return result;

406 static SystemError_t estop_update_led_indication(void) {
407  uint32_t current_time = HAL_Abstraction_GetTick();
408  SystemError_t result = SYSTEM_OK;
409 
410  switch (estop_context.state) {
411  case EMERGENCY_STOP_ARMED:
412  // Heartbeat indication (slow blink)
413  if ((current_time % (ESTOP_HEARTBEAT_TIME_MS * 2)) <
414  ESTOP_HEARTBEAT_TIME_MS) {
415  result = emergency_stop_set_led(true);
416  } else {
417  result = emergency_stop_set_led(false);
418  }
419  break;
420 
421  case EMERGENCY_STOP_TRIGGERED:
422  // Solid on when triggered
423  result = emergency_stop_set_led(true);
424  break;
425 
426  case EMERGENCY_STOP_RESET_PENDING:
427  // Fast blink during reset
428  if ((current_time % 200) < 100) { // 200ms period, 50% duty
429  result = emergency_stop_set_led(true);
430  } else {
431  result = emergency_stop_set_led(false);
432  }
433  break;
434 
435  case EMERGENCY_STOP_FAULT:
436  // Very fast blink for fault
437  if ((current_time % 100) < 50) { // 100ms period, 50% duty
438  result = emergency_stop_set_led(true);
439  } else {
440  result = emergency_stop_set_led(false);
441  }
442  break;
443 
444  default:
445  result = emergency_stop_set_led(false);
446  break;
447  }
448
