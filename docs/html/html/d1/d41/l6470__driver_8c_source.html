321  uint32_t *value) {
322  if (value == NULL) {
323  return ERROR_MOTOR_CONFIG_INVALID;
324  }

327  if (result != SYSTEM_OK) {
328  return result;

332  // Use simulation for parameter read if enabled
333  if (driver_state[motor_id].simulation_mode) {
334  simulation_error_t sim_result =
335  l6470_sim_read_register(register_addr, value);
336  return (sim_result == SIM_OK) ? SYSTEM_OK
337  : ERROR_MOTOR_COMMUNICATION_FAILED;
338  }

340 
341  uint8_t command = L6470_CMD_GET_PARAM | register_addr;
342 
343  // Prepare daisy-chain command
344  uint8_t tx_buffer[8] = {0};
345  uint8_t rx_buffer[8] = {0};
346 
347  if (motor_id == 0) {
348  // Command motor 1, NOP for motor 2
349  tx_buffer[0] = command;
350  tx_buffer[1] = 0x00;
351  tx_buffer[2] = 0x00;
352  tx_buffer[3] = 0x00;
353  tx_buffer[4] = L6470_CMD_NOP;
354  tx_buffer[5] = 0x00;
355  tx_buffer[6] = 0x00;
356  tx_buffer[7] = 0x00;
357  } else {
358  // NOP for motor 1, command motor 2
359  tx_buffer[0] = L6470_CMD_NOP;
360  tx_buffer[1] = 0x00;
361  tx_buffer[2] = 0x00;
362  tx_buffer[3] = 0x00;
363  tx_buffer[4] = command;
364  tx_buffer[5] = 0x00;
365  tx_buffer[6] = 0x00;
366  tx_buffer[7] = 0x00;
367  }
368 
369  result = l6470_spi_transaction(tx_buffer, rx_buffer, 8);
370  if (result != SYSTEM_OK) {
371  return result;
372  }
373 
374  // Extract parameter value from response
375  uint8_t param_bytes = 3; // Adjust based on register
376  uint8_t start_byte = (motor_id == 0) ? 0 : 4;
377 
378  l6470_unpack_parameter(&rx_buffer[start_byte], value, param_bytes);
379 
380  return SYSTEM_OK;

389 SystemError_t l6470_get_status(uint8_t motor_id, uint16_t *status) {
390  if (status == NULL) {
391  return ERROR_MOTOR_CONFIG_INVALID;
392  }
393 
394  SystemError_t result = l6470_validate_motor_id(motor_id);
395  if (result != SYSTEM_OK) {
396  return result;
397  }
398 
399 #if SIMULATION_ENABLED
400  // Use simulation for status read if enabled
401  if (driver_state[motor_id].simulation_mode) {
402  uint32_t status_value = 0;
403  simulation_error_t sim_result =
404  l6470_sim_read_register(L6470_REG_STATUS, &status_value);
405  if (sim_result == SIM_OK) {
406  *status = (uint16_t)(status_value & 0xFFFF);
407  return SYSTEM_OK;
408  }
409  return ERROR_MOTOR_COMMUNICATION_FAILED;
410  }
411 #endif
412 
413  uint8_t tx_data[2] = {L6470_CMD_GET_STATUS, 0x00};
414  uint8_t rx_data[2] = {0};
415 
416  // Assert CS
417  HAL_Abstraction_GPIO_Write(HAL_GPIO_PORT_A, 9, HAL_GPIO_STATE_RESET);
418 
419  // Small delay for L6470 setup time (from datasheet)
420  HAL_Abstraction_Delay(1);
421 
422  // SPI transaction using HAL abstraction
423  HAL_SPI_Transaction_t transaction = {.tx_data = tx_data,
424  .rx_data = rx_data,
425  .data_size = 2,
426  .timeout_ms = SPI_TIMEOUT_MS};
427  SystemError_t spi_result =
428  HAL_Abstraction_SPI_TransmitReceive(HAL_SPI_INSTANCE_2, &transaction);
429 
430  // Deassert CS
431  HAL_Abstraction_GPIO_Write(HAL_GPIO_PORT_A, 9, HAL_GPIO_STATE_SET);
432 
433  if (spi_result != ERROR_CODE_SUCCESS) {
434  spi_health.error_count++;
435  spi_health.last_error_time = HAL_Abstraction_GetTick();
436  spi_health.last_error_code = ERROR_SPI_TRANSMISSION_FAILED;
437  return ERROR_SPI_TRANSMISSION_FAILED;
438  }
439 
440  *status = (rx_data[0] << 8) | rx_data[1];
441 
442  // Update driver state
443  driver_state[motor_id].last_status = *status;
444 
445  // Check for L6470 error flags (per safety-systems.instructions.md)
446  if (*status & L6470_STATUS_UVLO) {
447  driver_state[motor_id].fault_count++;
448  return ERROR_MOTOR_UNDERVOLTAGE;
449  }
450  if (*status & L6470_STATUS_OCD) {
451  driver_state[motor_id].fault_count++;
452  return ERROR_MOTOR_OVERCURRENT;
453  }
454  if (*status & L6470_STATUS_TH_SD) {
455  driver_state[motor_id].fault_count++;
456  return ERROR_MOTOR_OVERHEAT;
457  }
458  if (*status & (L6470_STATUS_STEP_LOSS_A | L6470_STATUS_STEP_LOSS_B)) {
459  driver_state[motor_id].fault_count++;
460  return ERROR_MOTOR_STALL;
461  }
462 
463  return SYSTEM_OK;
464 }

629  uint16_t size) {

638  HAL_Abstraction_GPIO_Write(HAL_GPIO_PORT_A, 9, HAL_GPIO_STATE_RESET);
639  if (gpio_result != SYSTEM_OK) {
640  spi_health.error_count++;
641  spi_health.last_error_time = HAL_Abstraction_GetTick();
642  spi_health.last_error_code = ERROR_GPIO_WRITE_FAILED;
643  return ERROR_GPIO_WRITE_FAILED;

645 
646  // Small setup delay (per L6470 datasheet)
647  HAL_Abstraction_Delay(1);
648 
649  // Prepare SPI transaction using HAL abstraction
650  HAL_SPI_Transaction_t transaction = {.tx_data = tx_data,
651  .rx_data = rx_data,
652  .data_size = size,
653  .timeout_ms = SPI_TIMEOUT_MS};
654 
655  SystemError_t spi_result =
656  HAL_Abstraction_SPI_TransmitReceive(HAL_SPI_INSTANCE_2, &transaction);
657 
658  // Deassert CS (active low, so set high)
659  gpio_result =
660  HAL_Abstraction_GPIO_Write(HAL_GPIO_PORT_A, 9, HAL_GPIO_STATE_SET);
661  if (gpio_result != SYSTEM_OK) {

672  return ERROR_SPI_TRANSMISSION_FAILED;

686 static SystemError_t l6470_send_daisy_command(uint8_t motor1_cmd,
687  uint32_t motor1_data,
688  uint8_t motor2_cmd,
689  uint32_t motor2_data) {
690  uint8_t tx_buffer[8];
691  uint8_t rx_buffer[8];
692 
693  // Pack daisy-chain data (motor 1 first, motor 2 second)
694  tx_buffer[0] = motor1_cmd;
695  tx_buffer[1] = (motor1_data >> 16) & 0xFF;
696  tx_buffer[2] = (motor1_data >> 8) & 0xFF;
697  tx_buffer[3] = motor1_data & 0xFF;
698  tx_buffer[4] = motor2_cmd;
699  tx_buffer[5] = (motor2_data >> 16) & 0xFF;
700  tx_buffer[6] = (motor2_data >> 8) & 0xFF;
701  tx_buffer[7] = motor2_data & 0xFF;
702 
703  return l6470_spi_transaction(tx_buffer, rx_buffer, 8);
