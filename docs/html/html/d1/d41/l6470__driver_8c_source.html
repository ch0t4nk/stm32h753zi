331

336  if (*status & L6470_STATUS_UVLO) {
337  driver_state[motor_id].fault_count++;
338  return ERROR_MOTOR_UNDERVOLTAGE;

349  driver_state[motor_id].fault_count++;

351  }
352 
353  return SYSTEM_OK;

492 static SystemError_t l6470_send_daisy_command(uint8_t motor1_cmd, uint32_t motor1_data,
493  uint8_t motor2_cmd, uint32_t motor2_data) {
494  uint8_t tx_buffer[8];
495  uint8_t rx_buffer[8];
496 
497  // Pack daisy-chain data (motor 1 first, motor 2 second)
498  tx_buffer[0] = motor1_cmd;
499  tx_buffer[1] = (motor1_data >> 16) & 0xFF;
500  tx_buffer[2] = (motor1_data >> 8) & 0xFF;
501  tx_buffer[3] = motor1_data & 0xFF;
502  tx_buffer[4] = motor2_cmd;
503  tx_buffer[5] = (motor2_data >> 16) & 0xFF;
504  tx_buffer[6] = (motor2_data >> 8) & 0xFF;
505  tx_buffer[7] = motor2_data & 0xFF;
506 
507  return l6470_spi_transaction(tx_buffer, rx_buffer, 8);
508 }

634 SystemError_t l6470_run(uint8_t motor_id, bool direction, float speed) {
635  if (motor_id >= L6470_MAX_DEVICES) {
636  return ERROR_MOTOR_INVALID_ID;
637  }
638 
639  if (!l6470_initialized) {
640  return ERROR_MOTOR_INIT_FAILED;
641  }
642 
643  // Convert speed to L6470 format (step/tick)
644  // L6470 uses 250ns ticks, so steps/second = steps/tick * 4000000
645  uint32_t l6470_speed = (uint32_t)(speed / 4000000.0f * 0x10000000UL);
646 
647  // Clamp to valid range (0 to 0x3FF)
648  if (l6470_speed > 0x3FF) {
649  l6470_speed = 0x3FF;
650  }
651 
652  // Build RUN command with direction
653  uint8_t command = L6470_CMD_RUN;
654  if (direction) {
655  command |= 0x01; // Forward direction
656  }
657 
658  // Send command via daisy chain
659  if (motor_id == 0) {
660  // Motor 0: send command, Motor 1: NOP
661  return l6470_send_daisy_command(command, l6470_speed, L6470_CMD_NOP, 0);
662  } else {
663  // Motor 0: NOP, Motor 1: send command
664  return l6470_send_daisy_command(L6470_CMD_NOP, 0, command, l6470_speed);
665  }
