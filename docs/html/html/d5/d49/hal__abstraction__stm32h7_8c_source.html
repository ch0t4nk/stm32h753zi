540 uint32_t HAL_Abstraction_GetMicroseconds(void) {
541  // Use SysTick for microsecond timing
542  uint32_t tick = HAL_GetTick();
543  uint32_t systick_val = SysTick->VAL;
544  uint32_t systick_load = SysTick->LOAD;
545 
546  // Calculate microseconds within current millisecond
547  uint32_t us_in_tick = ((systick_load - systick_val) * 1000) / systick_load;
548 
549  return (tick * 1000) + us_in_tick;
550 }
