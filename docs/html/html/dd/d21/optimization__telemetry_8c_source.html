708  } else {

719 static SystemError_t telemetry_calculate_derivatives(
720  TelemetryContext_t *context, float current_position, float dt_seconds,
721  float *velocity_dps, float *acceleration_dps2) {
722  if (dt_seconds <= 0.0f || dt_seconds > 0.1f) {
723  // Invalid time delta - use last known values
724  *velocity_dps = context->last_velocity_dps;
725  *acceleration_dps2 = 0.0f;
726  return ERROR_INVALID_PARAMETER;
727  }
728 
729  // Calculate velocity using numerical derivative
730  float position_delta = current_position - context->last_position_degrees;
731 
732  // Handle position wraparound (360° -> 0°)
733  if (position_delta > 180.0f) {
734  position_delta -= 360.0f;
735  } else if (position_delta < -180.0f) {
736  position_delta += 360.0f;
737  }
738 
739  *velocity_dps = position_delta / dt_seconds;
740 
741  // Calculate acceleration using numerical derivative of velocity
742  float velocity_delta = *velocity_dps - context->last_velocity_dps;
743  *acceleration_dps2 = velocity_delta / dt_seconds;
744 
745  // Update context for next calculation
746  context->last_position_degrees = current_position;
747  context->last_velocity_dps = *velocity_dps;
748 
749  return SYSTEM_OK;
750 }
