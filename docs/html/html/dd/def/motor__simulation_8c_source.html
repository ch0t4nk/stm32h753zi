454 static void l6470_sim_update_motion(uint8_t motor_id, float delta_time_s) {
455  L6470_SimState_t *motor = &g_simulation.motors[motor_id];
456 
457  switch (motor->motion_state) {
458  case SIM_MOTOR_ACCELERATING:
459  if (motor->current_speed < motor->target_speed) {
460  motor->current_speed += motor->acceleration * delta_time_s;
461  if (motor->current_speed >= motor->target_speed) {
462  motor->current_speed = motor->target_speed;
463  motor->motion_state = SIM_MOTOR_CONSTANT_SPEED;
464  }
465  }
466  break;
467 
468  case SIM_MOTOR_DECELERATING:
469  if (motor->current_speed > 0.0f) {
470  motor->current_speed -= motor->deceleration * delta_time_s;
471  if (motor->current_speed <= 0.0f) {
472  motor->current_speed = 0.0f;
473  motor->motion_state = SIM_MOTOR_STOPPED;
474  motor->busy = false;
475  }
476  }
477  break;
478 
479  case SIM_MOTOR_CONSTANT_SPEED:
480  // Check if we need to start decelerating to reach target position
481  if (motor->target_position != motor->current_position) {
482  float distance_remaining = fabsf(
483  (float)(motor->target_position - motor->current_position));
484  float decel_distance =
485  (motor->current_speed * motor->current_speed) /
486  (2.0f * motor->deceleration);
487 
488  if (distance_remaining <= decel_distance) {
489  motor->motion_state = SIM_MOTOR_DECELERATING;
490  }
491  }
492  break;
493 
494  case SIM_MOTOR_STOPPED:
495  motor->current_speed = 0.0f;
496  motor->busy = false;
497  break;
498  }
499 
500  // Update position based on speed
501  float position_delta = motor->current_speed * delta_time_s;
502  if (motor->direction) {
503  motor->current_position += (int32_t)position_delta;
504  } else {
505  motor->current_position -= (int32_t)position_delta;
506  }
507 }

509 static void as5600_sim_update_position(uint8_t encoder_id,
510  float delta_time_s) {
511  AS5600_SimState_t *encoder = &g_simulation.encoders[encoder_id];
512  float elapsed_time_s =
513  (g_simulation.simulation_time_ms - encoder->pattern_start_time) /
514  1000.0f;
515 
516  switch (encoder->pattern) {
517  case SIM_PATTERN_LINEAR:
518  // Linear motion at constant velocity (degrees/second)
519  encoder->actual_position_deg +=
520  encoder->pattern_parameter * delta_time_s;
521  break;
522 
523  case SIM_PATTERN_SINUSOIDAL:
524  // Sinusoidal motion: amplitude * sin(2*pi*frequency*time)
525  encoder->actual_position_deg =
526  encoder->pattern_parameter *
527  sinf(2.0f * M_PI * 0.1f * elapsed_time_s);
528  break;
529 
530  case SIM_PATTERN_STATIC:
531  default:
532  // No motion
533  break;
534  }
535 
536  // Normalize to 0-360 degrees
537  encoder->actual_position_deg = fmodf(encoder->actual_position_deg, 360.0f);
538  if (encoder->actual_position_deg < 0.0f) {
539  encoder->actual_position_deg += 360.0f;
540  }
541

543  uint16_t angle_12bit =
544  (uint16_t)((encoder->actual_position_deg / 360.0f) * 4096.0f);
545  encoder->raw_angle = angle_12bit;
546  encoder->filtered_angle =
547  as5600_sim_add_noise(angle_12bit, encoder->noise_amplitude);

550 static uint16_t as5600_sim_add_noise(uint16_t angle, float noise_amplitude) {
551  if (noise_amplitude <= 0.0f) {
552  return angle;
553  }
554 
555  // Simple random noise generation
556  float noise = ((float)rand() / RAND_MAX - 0.5f) * 2.0f * noise_amplitude;
557  float angle_deg = (float)angle * 360.0f / 4096.0f;
558  angle_deg += noise;
559 
560  // Normalize and convert back
561  angle_deg = fmodf(angle_deg, 360.0f);
562  if (angle_deg < 0.0f)
563  angle_deg += 360.0f;
564 
565  return (uint16_t)((angle_deg / 360.0f) * 4096.0f);
566 }

585  break;

589  if (motor->direction) {
590  motor->status_register |= L6470_STATUS_DIR;
591  } else {
592  motor->status_register &= ~L6470_STATUS_DIR;
593  }
594

596  if (motor->busy) {
597  motor->status_register |= L6470_STATUS_BUSY;
598  } else {
599  motor->status_register &= ~L6470_STATUS_BUSY;
600  }
601

604  switch (motor->active_fault) {
605  case SIM_FAULT_OVERCURRENT:
606  motor->status_register |= L6470_STATUS_OCD;
607  break;
608  case SIM_FAULT_OVERHEAT:
609  motor->status_register |= L6470_STATUS_TH_SD;
610  break;
611  case SIM_FAULT_UNDERVOLTAGE:
612  motor->status_register |= L6470_STATUS_UVLO;
613  break;
614  case SIM_FAULT_STALL:
615  motor->status_register |= L6470_STATUS_STEP_LOSS_A;
616  break;
