94 SystemError_t watchdog_refresh(void) {
95  if (!watchdog_initialized || !watchdog_config.enabled) {
96  return SYSTEM_OK;
97  }
98 
99  uint32_t current_time = HAL_GetTick();
100  uint32_t elapsed = current_time - watchdog_config.last_refresh;
101 
102  // Check refresh interval validity
103  if (!is_refresh_interval_valid(elapsed)) {
104  consecutive_missed_refreshes++;
105  watchdog_config.missed_refresh_count++;
106 
107  // Check for excessive missed refreshes
108  if (consecutive_missed_refreshes > WATCHDOG_MISSED_KICK_MAX) {
109  return ERROR_SAFETY_WATCHDOG_MISSED;
110  }
111  } else {
112  consecutive_missed_refreshes = 0;
113  watchdog_warning_active = false;
114  }
115 
116  // Check if we're approaching timeout (warning)
117  if (elapsed > watchdog_config.warning_threshold_ms) {
118  watchdog_warning_active = true;
119  // Warning logged by caller (safety system)
120  }
121 
122  // Refresh Independent Watchdog
123  if (IWDG_ENABLE) {
124  HAL_IWDG_Refresh(&hiwdg);
125  }
126 
127  // Refresh Window Watchdog
128  if (WWDG_ENABLE) {
129  // WWDG refresh requires writing to counter register within window
130  HAL_WWDG_Refresh(&hwwdg);
131  }
132 
133  // Update tracking variables
134  last_refresh_interval = elapsed;
135  watchdog_config.last_refresh = current_time;
136  watchdog_config.refresh_count++;
137 
138  return SYSTEM_OK;
139 }

358 static void calculate_watchdog_timing(void) {
359  // Verify that SSOT timing parameters are reasonable
360 
361  // Check that kick interval is less than warning threshold
362  if (WATCHDOG_KICK_INTERVAL_MS >= WATCHDOG_LATE_KICK_MS) {
363  // Configuration error: adjust kick interval
364  // This should be fixed in SSOT configuration
365  }
366 
367  // Check that warning threshold is less than timeout
368  if (WATCHDOG_LATE_KICK_MS >= IWDG_TIMEOUT_MS) {
369  // Configuration error: timing hierarchy is incorrect
370  }
371 
372  // Calculate actual IWDG timeout based on hardware
373  // LSI frequency is typically ~32 kHz but can vary
374  // Actual timeout = (4 * 2^prescaler * reload) / LSI_freq
375  uint32_t prescaler_value = 4 << ((IWDG_PRESCALER >> IWDG_PR_PR_Pos) & 0x07);
376  uint32_t calculated_timeout = (4 * prescaler_value * IWDG_RELOAD_VALUE) / 32000; // Assume 32kHz LSI
377 
378  // Update configuration with calculated value if significantly different
379  if (abs((int32_t)calculated_timeout - (int32_t)IWDG_TIMEOUT_MS) > 10) {
380  // Log discrepancy (actual vs configured timeout)
381  // In production, this might warrant updating SSOT configuration
382  }
383 }
