67 SystemError_t motion_profile_generate_trapezoidal(MotionProfile_t *profile,
68  int32_t start_pos,
69  int32_t end_pos,
70  uint32_t max_vel,
71  uint32_t acceleration) {
72  if (profile == NULL) {
73  return ERROR_INVALID_PARAMETER;
74  }
75 
76  // Validate parameters
77  if (max_vel > MOTOR_MAX_SPEED || acceleration > MOTOR_MAX_ACCELERATION) {
78  return ERROR_MOTOR_PARAMETER_OUT_OF_RANGE;
79  }
80 
81  // Initialize profile
82  profile->type = PROFILE_TRAPEZOIDAL;
83  profile->start_position = start_pos;
84  profile->end_position = end_pos;
85  profile->max_velocity = max_vel;
86  profile->acceleration = acceleration;
87  profile->deceleration = acceleration; // Use same for deceleration
88  profile->current_phase = PROFILE_PHASE_ACCEL;
89  profile->start_time_ms = 0;
90 
91  // Calculate distances and times
92  int32_t total_distance = abs(end_pos - start_pos);
93  profile->direction = (end_pos > start_pos) ? 1 : -1;
94 
95  // Time to reach maximum velocity
96  float accel_time = (float)max_vel / acceleration;
97 
98  // Distance covered during acceleration
99  int32_t accel_distance = (acceleration * accel_time * accel_time) / 2;
100 
101  // Check if we can reach maximum velocity
102  if (2 * accel_distance <= total_distance) {
103  // Trapezoidal profile - we reach max velocity
104  profile->accel_time_ms = (uint32_t)(accel_time * 1000);
105  profile->accel_distance = accel_distance;
106  profile->decel_distance = accel_distance; // Symmetric
107  profile->const_vel_distance = total_distance - 2 * accel_distance;
108  profile->const_vel_time_ms =
109  (uint32_t)((float)profile->const_vel_distance / max_vel * 1000);
110  profile->decel_time_ms = profile->accel_time_ms;
111  profile->total_time_ms = profile->accel_time_ms +
112  profile->const_vel_time_ms +
113  profile->decel_time_ms;
114  profile->peak_velocity = max_vel;
115  } else {
116  // Triangular profile - never reach max velocity
117  float triangle_time = sqrtf((float)total_distance / acceleration);
118  profile->peak_velocity = (uint32_t)(acceleration * triangle_time);
119  profile->accel_time_ms = (uint32_t)(triangle_time * 1000);
120  profile->accel_distance = total_distance / 2;
121  profile->decel_distance = total_distance / 2;
122  profile->const_vel_distance = 0;
123  profile->const_vel_time_ms = 0;
124  profile->decel_time_ms = profile->accel_time_ms;
125  profile->total_time_ms = 2 * profile->accel_time_ms;
126  }
127 
128  return SYSTEM_OK;
129 }

234 static SystemError_t execute_trapezoidal_profile(MotionProfile_t *profile,
235  uint32_t elapsed_time_ms,
236  int32_t *target_pos,
237  uint32_t *target_vel) {
238  float elapsed_sec = (float)elapsed_time_ms / 1000.0f;
239  float accel_time_sec = (float)profile->accel_time_ms / 1000.0f;
240  float const_vel_time_sec = (float)profile->const_vel_time_ms / 1000.0f;
241 
242  if (elapsed_time_ms <= profile->accel_time_ms) {
243  // Acceleration phase
244  profile->current_phase = PROFILE_PHASE_ACCEL;
245  *target_vel = (uint32_t)(profile->acceleration * elapsed_sec);
246  float distance =
247  0.5f * profile->acceleration * elapsed_sec * elapsed_sec;
248  *target_pos =
249  profile->start_position + (int32_t)(distance * profile->direction);
250 
251  } else if (elapsed_time_ms <=
252  (profile->accel_time_ms + profile->const_vel_time_ms)) {
253  // Constant velocity phase
254  profile->current_phase = PROFILE_PHASE_CONST_VEL;
255  *target_vel = profile->peak_velocity;
256  float const_vel_elapsed = elapsed_sec - accel_time_sec;
257  float distance = profile->accel_distance +
258  (profile->peak_velocity * const_vel_elapsed);
259  *target_pos =
260  profile->start_position + (int32_t)(distance * profile->direction);
261 
262  } else {
263  // Deceleration phase
264  profile->current_phase = PROFILE_PHASE_DECEL;
265  float decel_elapsed =
266  elapsed_sec - accel_time_sec - const_vel_time_sec;
267  *target_vel = (uint32_t)(profile->peak_velocity -
268  (profile->deceleration * decel_elapsed));
269 
270  if (*target_vel > profile->peak_velocity) {
271  *target_vel = 0; // Handle underflow
272  }
273 
274  float decel_distance =
275  profile->peak_velocity * decel_elapsed -
276  0.5f * profile->deceleration * decel_elapsed * decel_elapsed;
277  float total_distance = profile->accel_distance +
278  profile->const_vel_distance + decel_distance;
279  *target_pos = profile->start_position +
280  (int32_t)(total_distance * profile->direction);
281  }
282 
283  return SYSTEM_OK;
284 }

289 static SystemError_t execute_scurve_profile(MotionProfile_t *profile,
290  uint32_t elapsed_time_ms,
291  int32_t *target_pos,
292  uint32_t *target_vel) {
293  // Simplified S-curve implementation
294  // For full implementation, would need to handle all 7 phases of S-curve
295 
296  float elapsed_sec = (float)elapsed_time_ms / 1000.0f;
297  float jerk_time_sec = (float)profile->jerk_time_ms / 1000.0f;
298 
299  if (elapsed_time_ms <= profile->jerk_accel_time_ms) {
300  // Jerk acceleration phase
301  profile->current_phase = PROFILE_PHASE_JERK_ACCEL;
302  float jerk_accel = 0.5f * profile->jerk * elapsed_sec * elapsed_sec;
303  *target_vel = (uint32_t)(jerk_accel);
304  float distance = (1.0f / 6.0f) * profile->jerk * elapsed_sec *
305  elapsed_sec * elapsed_sec;
306  *target_pos =
307  profile->start_position + (int32_t)(distance * profile->direction);
308 
309  } else {
310  // For simplicity, fall back to linear acceleration after jerk phase
311  // Full S-curve would have 7 distinct phases
312  float remaining_accel_time = elapsed_sec - jerk_time_sec;
313  float jerk_velocity =
314  0.5f * profile->jerk * jerk_time_sec * jerk_time_sec;
315  *target_vel = (uint32_t)(jerk_velocity +
316  profile->acceleration * remaining_accel_time);
317 
318  // Simplified position calculation
319  float jerk_distance = (1.0f / 6.0f) * profile->jerk * jerk_time_sec *
320  jerk_time_sec * jerk_time_sec;
321  float linear_distance = jerk_velocity * remaining_accel_time +
322  0.5f * profile->acceleration *
323  remaining_accel_time *
324  remaining_accel_time;
325  *target_pos =
326  profile->start_position +
327  (int32_t)((jerk_distance + linear_distance) * profile->direction);
328  }
329 
330  return SYSTEM_OK;
331 }

428 SystemError_t motion_profile_synchronize(uint8_t *motor_ids,
429  MotionProfile_t *profiles,
430  uint8_t motor_count) {
431  if (motor_ids == NULL || profiles == NULL || motor_count == 0 ||
432  motor_count > MAX_MOTORS) {
433  return ERROR_INVALID_PARAMETER;
434  }
435 
436  // Find the longest profile time
437  uint32_t max_time = 0;
438  for (uint8_t i = 0; i < motor_count; i++) {
439  if (profiles[i].total_time_ms > max_time) {
440  max_time = profiles[i].total_time_ms;
441  }
442  }
443 
444  // Scale all profiles to match the longest time
445  for (uint8_t i = 0; i < motor_count; i++) {
446  if (profiles[i].total_time_ms < max_time) {
447  // Reduce velocity to extend time
448  float scale_factor = (float)profiles[i].total_time_ms / max_time;
449  profiles[i].max_velocity =
450  (uint32_t)(profiles[i].max_velocity * scale_factor);
451  profiles[i].peak_velocity =
452  (uint32_t)(profiles[i].peak_velocity * scale_factor);
453 
454  // Recalculate timing with new velocity
455  motion_profile_generate_trapezoidal(
456  &profiles[i], profiles[i].start_position,
457  profiles[i].end_position, profiles[i].max_velocity,
458  profiles[i].acceleration);
459  }
460  }
461 
462  // Start all profiles simultaneously
463  uint32_t sync_start_time = HAL_Abstraction_GetTick();
464  for (uint8_t i = 0; i < motor_count; i++) {
465  profiles[i].start_time_ms = sync_start_time;
466  motion_profile_start(motor_ids[i], &profiles[i]);
467  }
468 
469  return SYSTEM_OK;
