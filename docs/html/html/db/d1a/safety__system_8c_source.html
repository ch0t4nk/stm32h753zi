376 SystemError_t handle_safety_violation(MonitorChannel_t channel, float value) {
377  SystemError_t result = SYSTEM_OK;
378 
379  switch (channel) {
380  case MONITOR_MOTOR1_CURRENT:
381  case MONITOR_MOTOR2_CURRENT:
382  // Overcurrent: Stop motor immediately
383  {
384  uint8_t motor_id = (channel == MONITOR_MOTOR1_CURRENT) ? 0 : 1;
385  // L6470 Integration: Immediate stop for overcurrent protection
386  SystemError_t motor_stop_result = l6470_hard_stop(motor_id);
387  if (motor_stop_result != SYSTEM_OK) {
388  // Log motor stop failure for diagnostics
389  }
390  log_safety_event(SAFETY_EVENT_OVERCURRENT_STOP, motor_id,
391  *(uint32_t *)&value);
392  result = execute_emergency_stop(ESTOP_SOURCE_OVERCURRENT);
393  }
394  break;
395 
396  case MONITOR_MOTOR1_SPEED:
397  case MONITOR_MOTOR2_SPEED:
398  // Overspeed: Reduce speed or stop
399  {
400  uint8_t motor_id = (channel == MONITOR_MOTOR1_SPEED) ? 0 : 1;
401  // L6470 Integration: Controlled deceleration for overspeed
402  SystemError_t motor_stop_result = l6470_soft_stop(motor_id);
403  if (motor_stop_result != SYSTEM_OK) {
404  // If soft stop fails, use hard stop for safety
405  l6470_hard_stop(motor_id);
406  }
407  log_safety_event(SAFETY_EVENT_OVERSPEED_STOP, motor_id,
408  *(uint32_t *)&value);
409  }
410  break;
411 
412  case MONITOR_MOTOR1_POSITION:
413  case MONITOR_MOTOR2_POSITION:
414  // Position limit: Stop motor and prevent further movement
415  {
416  uint8_t motor_id = (channel == MONITOR_MOTOR1_POSITION) ? 0 : 1;
417  // L6470 Integration: Immediate stop for position limit violation
418  SystemError_t motor_stop_result = l6470_hard_stop(motor_id);
419  if (motor_stop_result != SYSTEM_OK) {
420  // Log motor stop failure for diagnostics
421  }
422  log_safety_event(SAFETY_EVENT_POSITION_LIMIT_STOP, motor_id,
423  *(uint32_t *)&value);
424  result = execute_emergency_stop(ESTOP_SOURCE_POSITION_LIMIT);
425  }
426  break;
427 
428  case MONITOR_SYSTEM_TEMPERATURE:
429  // Overtemperature: Throttle or stop system
430  log_safety_event(SAFETY_EVENT_TEMPERATURE_WARNING, 0,
431  *(uint32_t *)&value);
432  if (value > MOTOR_TEMP_SHUTDOWN_C) {
433  result = execute_emergency_stop(ESTOP_SOURCE_OVERHEAT);
434  }
435  break;
436 
437  case MONITOR_SUPPLY_VOLTAGE:
438  // Power supply fault: Emergency stop
439  result = execute_emergency_stop(ESTOP_SOURCE_FAULT_CASCADE);
440  break;
441 
442  default:
443  // Unknown channel: Log warning
444  log_safety_event(SAFETY_EVENT_WARNING, channel, *(uint32_t *)&value);
445  break;
446  }
447 
448  return result;
449 }
